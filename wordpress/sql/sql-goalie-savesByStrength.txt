DROP TABLE IF EXISTS wp_tsa_goalie_savesbystrength;

CREATE TABLE wp_tsa_goalie_savesbystrength (
    gameDate DATE,
    gameId BIGINT,
    goalieFullName VARCHAR(150),
    playerId BIGINT,
    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),
    shootsCatches VARCHAR(10),
    gamesPlayed INT,
    gamesStarted INT,
    wins INT,
    losses INT,
    otLosses INT,
    shotsAgainst INT,
    saves INT,
    goalsAgainst INT,
    savePct DECIMAL(10,4),
    evShotsAgainst INT,
    evSaves INT,
    evGoalsAgainst INT,
    evSavePct DECIMAL(10,4),
    ppShotsAgainst INT,
    ppSaves INT,
    ppGoalsAgainst INT,
    ppSavePct DECIMAL(10,4),
    shShotsAgainst INT,
    shSaves INT,
    shGoalsAgainst INT,
    shSavePct DECIMAL(10,4),
    lastName VARCHAR(100),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_goalieFullName (goalieFullName)
);

DELETE FROM wp_tsa_goalie_savesbystrength
WHERE goalieFullName = 'goalieFullName';

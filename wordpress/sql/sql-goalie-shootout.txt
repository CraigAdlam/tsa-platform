DROP TABLE IF EXISTS wp_tsa_goalie_shootout;

CREATE TABLE wp_tsa_goalie_shootout (
    gameDate DATE,
    gameId BIGINT,
    goalieFullName VARCHAR(150),
    playerId BIGINT,
    homeRoad VARCHAR(10),
    teamAbbrev VARCHAR(10),
    opponentTeamAbbrev VARCHAR(10),
    shootsCatches VARCHAR(10),
    gamesPlayed INT,
    shootoutWins INT,
    shootoutLosses INT,
    shootoutShotsAgainst INT,
    shootoutGoalsAgainst INT,
    shootoutSaves INT,
    shootoutSavePct DECIMAL(10,4),
    careerShootoutGamesPlayed INT,
    careerShootoutWins INT,
    careerShootoutLosses INT,
    careerShootoutShotsAgainst INT,
    careerShootoutGoalsAllowed INT,
    careerShootoutSaves INT,
    careerShootoutSavePct DECIMAL(10,4),
    lastName VARCHAR(100),

    INDEX idx_gameDate (gameDate),
    INDEX idx_gameId (gameId),
    INDEX idx_playerId (playerId),
    INDEX idx_teamAbbrev (teamAbbrev),
    INDEX idx_opponentTeamAbbrev (opponentTeamAbbrev),
    INDEX idx_homeRoad (homeRoad),
    INDEX idx_goalieFullName (goalieFullName)
);

DELETE FROM wp_tsa_goalie_shootout
WHERE goalieFullName = 'goalieFullName';
